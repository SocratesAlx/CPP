#include <stdio.h>
#include <string.h>

int main() {
    char ben_array[20]; 

    printf("Enter a string you want reversed: ");
    fgets(ben_array, sizeof(ben_array), stdin); 
    
    int length = strlen(ben_array); 
    
    printf("The reversed string is: ");
    for (int i = length - 1; i >= 0; i--) {
        printf("%c", ben_array[i]); 
    }
    
    return 0;
}
